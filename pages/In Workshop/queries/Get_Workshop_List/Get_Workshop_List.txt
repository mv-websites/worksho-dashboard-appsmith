SELECT calls.Call_Ref, 
	call_status.Call_Status_Description, 
	equipment.Description ,
	concat(equipment.Make, ' | ', equipment.Model) as Make_Model,
	client.Co_Name ,
	calls.Order_No, 
	calls.Date_Received, 
	call_more.CALLM_Data119 as com_sales_val,
	call_more.CALLM_Data118 as wksp_proposed_date,
	calls.Problem,
	calls.Actual_Labour_Cost,
	calls.Actual_Parts_Cost,
	calls.Related_Job,
	calls.Scheduled_Date_Time_End,
	calls.Call_Next_ETA
FROM SICLOPS_LIVE_MVL.dbo.vw_LAYD_Calls calls
LEFT JOIN SICLOPS_LIVE_MVL.dbo.vw_LAYD_LU_Call_Status call_status
	ON calls.Last_Event_Status = call_status.Call_Status_Code 
LEFT JOIN SICLOPS_LIVE_MVL.dbo.vw_LAYD_Equipment equipment
	ON calls.Link_to_Equipment = equipment.Equipment_Code 
LEFT JOIN SICLOPS_LIVE_MVL.dbo.vw_LAYD_Clients client
	ON equipment.Link_to_Client = client.Client_Ref 
LEFT JOIN SICLOPS_LIVE_MVL.dbo.vw_LAYD_Call_More call_more
	ON call_more.CALLM_Link_to_Call = calls.Call_Ref 
WHERE	Last_Event_Status {{Status_Select.selectedOptionValue === "" ? `IN ('XR', 'M1', 'PK', 'HV', 'PB', 'PD', 'RB', 'RH', 'VF', 'WC', 'WS')` : `= '${Status_Select.selectedOptionValue}'` }}
{{Customer_Select.selectedOptionValue === ""? `` : `AND client.Co_Name = '${Customer_Select.selectedOptionValue}'`}}
{{Description_Select.selectedOptionValue === ""? `` : `AND equipment.Description = '${Description_Select.selectedOptionValue}'`}}
