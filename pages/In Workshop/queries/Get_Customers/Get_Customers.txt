SELECT DISTINCT(client.Co_Name) as label, client.Co_Name as value
FROM SICLOPS_LIVE_MVL.dbo.vw_LAYD_Calls calls
LEFT JOIN SICLOPS_LIVE_MVL.dbo.vw_LAYD_Equipment equipment
	ON calls.Link_to_Equipment = equipment.Equipment_Code 
LEFT JOIN SICLOPS_LIVE_MVL.dbo.vw_LAYD_Clients client
	ON equipment.Link_to_Client = client.Client_Ref 
WHERE Last_Event_Status IN ('XR', 'M1', 'PK', 'HV', 'PB', 'PD', 'RB', 'RH', 'VF', 'WC', 'WS');
