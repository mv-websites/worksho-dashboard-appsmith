SELECT
    calls.Last_Event_Status,
    call_status.Call_Status_Description,
    COUNT(*) AS Status_Count
FROM SICLOPS_LIVE_MVL.dbo.vw_LAYD_Calls calls
LEFT JOIN SICLOPS_LIVE_MVL.dbo.vw_LAYD_LU_Call_Status call_status
    ON calls.Last_Event_Status = call_status.Call_Status_Code 
LEFT JOIN SICLOPS_LIVE_MVL.dbo.vw_LAYD_Equipment equipment
    ON calls.Link_to_Equipment = equipment.Equipment_Code 
LEFT JOIN SICLOPS_LIVE_MVL.dbo.vw_LAYD_Clients client
    ON equipment.Link_to_Client = client.Client_Ref 
LEFT JOIN SICLOPS_LIVE_MVL.dbo.vw_LAYD_Call_More call_more
    ON call_more.CALLM_Link_to_Call = calls.Call_Ref 
WHERE calls.Last_Event_Status IN ('XR', 'M1', 'PK', 'HV', 'PB', 'PD', 'RB', 'RH', 'VF', 'WC', 'WS', 'AV')
GROUP BY
    calls.Last_Event_Status,
    call_status.Call_Status_Description
ORDER BY
    calls.Last_Event_Status;
